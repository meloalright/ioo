> 大致思路: 动态规划的基本思想

> "每当你想要买到更多 你总会付出更多 买还是不买的收益最佳策略 总是相当于 
(什么都不做的收益 与 强行买了该商品后剩下的资源的最佳收益+该商品的收益) 的最大值"

`chess_rush_dp.py`

```py
def chess_rush_dp(item_list, max_size):

    # length
    l = len(item_list) + 1

    # 2D List
    dp = [[0 for j in range(max_size + 1)] for i in range(l)]

    for number in range(1, l):
        for w in range(1, max_size + 1):
            this_item = item_list[number - 1]
            if (w < this_item["cost"]):
                dp[number][w] = dp[number-1][w]
                print_2d_list(dp)
                continue
            dp[number][w] = max(dp[number-1][w], dp[number-1]
                                [w-this_item["cost"]]+this_item["value"])
            print_2d_list(dp)

    # final
    return dp[l-1][max_size]
```

`print_2d_list`

```py
def print_2d_list(two_d_list):
    '''

    [0,0,0,0,0]
    [0,1,1,1,x]
    [0,x,x,x,x]
    [0,x,x,x,x]
    [0,x,x,x,x]

    >>>

    | 0 | 0 | 0 | 0 | 0 |
    | 0 | 1 | x | x | x |
    | 0 | 1 | x | x | x |
    | 0 | 1 | x | x | x |
    | 0 | x | x | x | x |

    '''
    print("-" * 23 + " 2D List " + "-" * 23)
    columns = len(two_d_list)  # 列数 (宽度)

    rows = len(two_d_list[0])  # 行数 (高度)
    row_index = 0

    while row_index < rows:
        this_row = [two_d_list[i][row_index] for i in range(columns)]
        pstr = "|"
        for it in this_row:
            s = str(it)
            pstr += ("  " + s + (" " * (3 - len(s))) + "|")
        print(pstr)
        row_index += 1
```
